def initialize(raw)
  @object = case raw
  when self.class::RAW
    raw
  when ::String
    begin
      ::JSON.parse(raw)
    rescue ::JSON::ParserError
      raw
    end
  when self.class
    raw.object
  else
    ::Kernel.raise ::TypeError,
      "%s does not accept %s" % [self.class, raw.inspect]
  end
end
