def initialize(raw)
  @object = case raw
  when self.class::RAW then raw
  when ::String        then ::JSON.parse(raw)
  when self.class      then raw.object
  else
    ::Kernel.raise ::TypeError, 
      "%s does not accept %s" % [self.class, raw.inspect]
  end
end
